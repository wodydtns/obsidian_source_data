public static void main(String[] args) {  
    int[] array = {7, 5, 9, 0, 3, 1, 6, 2, 4, 8};  
    int length = array.length;  
  
    for (int i = 0; i < array.length; i++) {  
        for (int j = 0; j < length - i - 1; j++) {  
            if(array[j] > array[j+1]){  
                int temp = array[j];  
                array[j] = array[j+1];  
                array[j+1] = temp;  
            }  
        }  
        }
